country_event = {
	id = aue_flavor.2
	title = aue_flavor.2.t
	desc = aue_flavor.2.d
	picture = ORTHODOX_eventPicture

	is_triggered_only = yes

	major = yes

	option = {#monarchy
		name = aue_flavor.2.a
		if = {
			limit = {
				NOT = { government = monarchy }
			}
			change_government = monarchy
		}
		add_government_reform = european_hedgemony
	}
	option = {#republic
		name = aue_flavor.2.b
		if = {
			limit = {
				NOT = { government = republic }
			}
			change_government = republic
		}
		add_government_reform = european_hedgemony
	}
	after = {
		country_event = { id = aue_flavor.3 }
	}
}
